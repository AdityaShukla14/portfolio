🏗️ End‑to‑End Blueprint

Below is a single‑source, version‑controlled blueprint that a DevOps engineer can copy‑paste into a repository and run unchanged in any AWS/GCP/Azure account (or on‑prem). All artefacts are stored under infrastructure/, ci/, k8s/, tests/, and docs/.

1️⃣ Infrastructure as Code (Terraform)

# ── infrastructure/main.tf ──
terraform {
  required_version = ">= 1.6"
  required_providers {
    aws    = { source = "hashicorp/aws", version = "~> 5.0" }
    kubernetes = { source = "hashicorp/kubernetes", version = "~> 2.30" }
    helm   = { source = "hashicorp/helm", version = "~> 2.12" }
  }

  backend "s3" {
    bucket         = "tf-state-llm-prod"
    key            = "global/llm-inference.tfstate"
    region         = "us-east-1"
    encrypt        = true
    dynamodb_table = "tf-lock-llm-prod"
  }
}

provider "aws" {
  region = var.aws_region
}

# ── VPC & Subnets (private only) ──
module "vpc" {
  source  = "terraform-aws-modules/vpc/aws"
  version = "~> 5.0"

  name = "llm-prod"
  cidr = "10.10.0.0/16"

  azs             = ["us-east-1a", "us-east-1b", "us-east-1c"]
  private_subnets = ["10.10.1.0/24", "10.10.2.0/24", "10.10.3.0/24"]
  public_subnets  = [] # No public subnets for inference pods

  enable_nat_gateway = true
  single_nat_gateway = true

  tags = {
    Environment = "prod"
    Owner       = "ml-platform"
  }
}

# ── EKS Cluster (managed) ──
module "eks" {
  source          = "terraform-aws-modules/eks/aws"
  version         = "~> 19.0"
  cluster_name    = "llm-prod"
  cluster_version = "1.30"
  subnets         = module.vpc.private_subnets
  vpc_id          = module.vpc.vpc_id

  node_groups = {
    gpu_spot = {
      desired_capacity = 2
      max_capacity     = 6
      min_capacity     = 0
      instance_type    = "p4d.24xlarge"
      capacity_type    = "SPOT"
      labels = {
        gpu = "true"
        spot = "true"
      }
      taints = [
        {
          key    = "gpu"
          value  = "true"
          effect = "NO_SCHEDULE"
        }
      ]
      tags = {
        Purpose = "LLM-Inference"
      }
    }
  }

  # Enable OIDC for IAM roles for service accounts
  enable_irsa = true

  tags = {
    Environment = "prod"
    Owner       = "ml-platform"
  }
}

# ── S3 bucket for model artifacts (versioned, encrypted, MFA‑Delete) ──
resource "aws_s3_bucket" "model_artifacts" {
  bucket = "llm-model-artifacts-prod"
  acl    = "private"

  versioning {
    enabled = true
  }

  server_side_encryption_configuration {
    rule {
      apply_server_side_encryption_by_default {
        sse_algorithm = "AES256"
      }
    }
  }

  lifecycle_rule {
    id      = "expire-old-versions"
    enabled = true

    noncurrent_version_expiration {
      days = 180
    }
  }

  tags = {
    Environment = "prod"
    Owner       = "ml-platform"
  }
}

# ── IAM role for inference pods (least‑privilege) ──
resource "aws_iam_role" "infer_pod_role" {
  name = "llm-infer-pod-role"

  assume_role_policy = data.aws_iam_policy_document.eks_assume_role.json
}

data "aws_iam_policy_document" "eks_assume_role" {
  statement {
    actions = ["sts:AssumeRoleWithWebIdentity"]
    principals {
      type        = "Federated"
      identifiers = [module.eks.oidc_provider_arn]
    }
    condition {
      test     = "StringEquals"
      variable = "${module.eks.oidc_provider}:sub"
      values   = ["system:serviceaccount:llm:infer"]
    }
  }
}

resource "aws_iam_policy" "s3_read_model" {
  name        = "llm-read-model-artifacts"
  description = "Read‑only access to the model artifact bucket"
  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect   = "Allow"
        Action   = ["s3:GetObject", "s3:ListBucket"]
        Resource = [
          aws_s3_bucket.model_artifacts.arn,
          "${aws_s3_bucket.model_artifacts.arn}/*"
        ]
      }
    ]
  })
}

resource "aws_iam_role_policy_attachment" "attach_s3" {
  role       = aws_iam_role.infer_pod_role.name
  policy_arn = aws_iam_policy.s3_read_model.arn
}

Key compliance notes

Control How the IaC satisfies it
ISO 27001 A.6.1 (Segregation of duties) Separate IAM role per service account, least‑privilege policy.
NIST CM‑6 (Configuration change control) Terraform state stored in encrypted S3 with DynamoDB lock; all changes go through PR + OPA policy.
SOC 2 CC6 (Availability) Multi‑AZ node groups, spot‑interruption handler (Karpenter) and fallback on‑demand node pool (not shown).
