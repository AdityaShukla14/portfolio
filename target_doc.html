1.3 CI Validation Step

# .github/workflows/validate-input.yml
name: Validate Request Payload
on:
  pull_request:
    paths:
      - 'requests/**/*.json'

jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Install ajv-cli
        run: npm i -g ajv-cli
      - name: Validate JSON payloads
        run: |
          for f in requests/**/*.json; do
            ajv-cli validate -s schema/request-schema.json -d "$f" --strict || exit 1
          done

Result: Any schema violation blocks the PR merge and posts a detailed comment (see Error Handling).

3.2 Implementation (excerpt)

# weight.py
import math

# Default weight configuration – can be overridden via env vars or a YAML file.
DEFAULT_WEIGHTS = {"A": 0.30, "B": 0.20, "C": 0.25, "D": 0.25}

def _score_availability(payload):
    sla = payload["constraints"].get("sla_uptime", "99.9")
    mapping = {"99.9": 90, "99.99": 95, "99.999": 99}
    return mapping.get(sla, 80)

def _score_business(payload):
    budget = payload["constraints"].get("budget_usd", 0)
    # Simple inverse relationship: higher budget → lower score (cost pressure)
    return max(0, 100 - math.log1p(budget) * 10)

def _score_compliance(payload):
    comps = payload["constraints"].get("compliance", [])
    return 100 if not comps else 100 - len(comps) * 10

def _score_deployment(payload):
    # Priority 5 (critical) gets higher score.
    return 70 + payload["priority"] * 6

def compute_score(payload: dict) -> dict:
    scores = {
        "A": _score_availability(payload),
        "B": _score_business(payload),
        "C": _score_compliance(payload),
        "D": _score_deployment(payload),
    }
    total = round(sum(scores[l] * DEFAULT_WEIGHTS[l] for l in scores))
    scores["total"] = total
    return scores

import pytest
from weight import compute_score

@pytest.fixture
def sample_payload():
    return {
        "request_id": "REQ123",
        "description": "Need 99.99% uptime for payment service.",
        "priority": 5,
        "constraints": {"sla_uptime": "99.99", "budget_usd": 5000, "compliance": ["PCI-DSS"]},

    }

def test_compute_score(sample_payload):
    scores = compute_score(sample_payload)
    assert scores["A"] == 95
    assert scores["C"] == 90
    assert scores["total"] > 80
